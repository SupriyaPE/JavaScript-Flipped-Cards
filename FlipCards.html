Promise Example


let promise = new Promise((resolve, reject) => {
  let dataFound = true;

  if (dataFound) {
    resolve("Data loaded successfully!");
  } else {
    reject("Data not found!");
  }
});

promise
  .then((msg) => console.log(msg))
  .catch((err) => console.log(err));
